k21_flt <- k21 %>% filter(S1 %in% list5per$SampleID) %>%
  filter(S2 %in% list5per$SampleID)

k21_flt <- as.matrix(k21_flt %>% 
                       pivot_wider(names_from = S2, values_from = dist) %>% 
                       column_to_rownames(var="S1")
)
